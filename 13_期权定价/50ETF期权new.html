def blackScholes(s,x,t,vol,r):
    d1=(np.log(s/float(x))+(r+0.5*(vol**2))*t)/(np.sqrt(t)*(vol))
    d2=d1-vol*np.sqrt(t)
    ret=s*norm.cdf(d1)-x*(np.exp(-r*t)*norm.cdf(d2))
    return ret

opt_pred=blackScholes(spot,strike,t,sigma,r)

def BinomalTree(sigma,r,steps,spot,dt):
#     dt=1/252 # 每一步的步长（步长为一个交易日）
    r=r/100 # 除以100去掉百分号
    up = np.exp((r  - 0.5*sigma*sigma)*dt + sigma*np.sqrt(dt))
    down = np.exp((r  - 0.5*sigma*sigma)*dt - sigma*np.sqrt(dt))

    discount = np.exp(-r*dt) #dt表示为日度波动率
    steps=int(steps)
    
#     print(up**365,down**365,steps,discount**365)
    stock=np.zeros((steps,steps))
    stock[0][0]=spot
    for i in range(1,steps):
        for j in range(0,i+1):
            if j==0:
                stock[i][j]=stock[i-1][j]*down
            else:
                stock[i][j]=stock[i-1][j-1]*up

    opt_tree=(stock-strike).clip(min=0) # 每一个节点行权后的收益
    opt_price_tree=opt_tree
    for i in range(steps-2,-1,-1):
        for j in range(0,i+1):
            # 逆推得到初始点的期权价格
            opt_price_tree[i][j]=(opt_price_tree[i+1][j]+opt_price_tree[i+1][j+1])/2*discount

    return opt_price_tree

# 计算各个时间点的期权预测价格
